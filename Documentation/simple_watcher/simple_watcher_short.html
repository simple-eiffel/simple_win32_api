note
	description: "[
		SCOOP-compatible file system watcher.
		Uses Win32 ReadDirectoryChangesW via C wrapper.
	]"
	author: "Larry Rix"
	date: "$Date$"
	revision: "$Revision$"

class interface
	SIMPLE_WATCHER

create 
	make

feature -- Constants: Watch Flags

	Watch_file_name: INTEGER_32 = 1
			-- Watch for file name changes (create, delete, rename).

	Watch_dir_name: INTEGER_32 = 2
			-- Watch for directory name changes.

	Watch_attributes: INTEGER_32 = 4
			-- Watch for attribute changes.

	Watch_size: INTEGER_32 = 8
			-- Watch for size changes.

	Watch_last_write: INTEGER_32 = 16
			-- Watch for last write time changes.

	Watch_security: INTEGER_32 = 32
			-- Watch for security descriptor changes.

	Watch_all: INTEGER_32 = 63
			-- Watch for all changes.
	
feature -- Constants: Event Types

	Event_file_added: INTEGER_32 = 1
			-- A file was added.

	Event_file_removed: INTEGER_32 = 2
			-- A file was removed.

	Event_file_modified: INTEGER_32 = 3
			-- A file was modified.

	Event_file_renamed: INTEGER_32 = 4
			-- A file was renamed.
	
feature -- Status

	is_valid: BOOLEAN
			-- Is the watcher handle valid?

	is_watching: BOOLEAN
			-- Is the watcher actively watching?

	watch_path: detachable STRING_8
			-- The path being watched.

	last_error: detachable STRING_32
			-- Error message from last failed operation.
	
feature -- Operations

	start
			-- Start watching for changes.
		require
			valid: is_valid

	poll: detachable SIMPLE_WATCH_EVENT
			-- Check for a change event without blocking.
			-- Returns Void if no event is pending.
		require
			valid: is_valid
			watching: is_watching

	wait (a_timeout_ms: INTEGER_32): detachable SIMPLE_WATCH_EVENT
			-- Wait for a change event with timeout (milliseconds).
			-- Use -1 for infinite wait.
			-- Returns Void on timeout.
		require
			valid: is_valid
			watching: is_watching

	close
			-- Stop watching and release resources.
		ensure
			closed: handle = default_pointer
	
feature -- Status Report

	last_start_succeeded: BOOLEAN
			-- Did the last start operation succeed?
	
end -- class SIMPLE_WATCHER
